            if details.get("broad_history_found"):
                with st.expander(f"📚 Histórico Bruto Encontrado ({len(details['broad_history_found'])} registros)"):
                    for i, failure in enumerate(details["broad_history_found"]):
                        st.markdown(f"**Análise Histórica {i+1}**")
                        
                        # Usando as funções para formatar e exibir cada seção, filtrando os vazios.
                        description = failure.get("Descrição do Problema", "N/A")
                        st.markdown(f"  - **Descrição:** *{description}*")
                        
                        root_cause_list = [rc.get('Causa') for rc in failure.get('Causa Raiz', []) if rc.get('Causa')]
                        root_cause = ", ".join(root_cause_list) if root_cause_list else "N/A"
                        st.markdown(f"  - **Causa Raiz Concluída:** *{root_cause}*")

                        ishikawa_formatted = format_ishikawa(failure.get("Ishikawa"))
                        if ishikawa_formatted != "No causes registered.":
                             st.markdown(f"  - **Ishikawa (Resumo):**\n```\n{ishikawa_formatted}\n```")

                        five_whys_formatted = format_5whys(failure.get("5 Porquês"))
                        if five_whys_formatted != "No 'why' registered.":
                            st.markdown(f"  - **5 Porquês (Resumo):**\n```\n{five_whys_formatted}\n```")
                        
                        action_plan_formatted = format_list(failure.get("Plano de Ação"))
                        if action_plan_formatted != "No actions registered.":
                            st.markdown(f"  - **Plano de Ação:** *{action_plan_formatted}*")
                        
                        st.divider()
